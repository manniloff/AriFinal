##M3UA
#Create AS
m3ua as create <as-name> <AS | SGW | IPSP> mode <SE | DE> ipspType <client | server> rc <routing-context> traffic-mode <traffic mode[loadshare | override]> <network-appearance>
m3ua as create AS1 AS mode SE ipspType client rc 100 traffic-mode loadshare

# Create ASP
m3ua asp create ASP1 unifun1
m3ua asp create ASP2 unifun2
m3ua asp create ASP3 unifun3
m3ua asp create ASP4 unifun4

# Add ASP to AS
m3ua as add AS1 ASP1
m3ua as add AS1 ASP2
m3ua as add AS1 ASP3
m3ua as add AS1 ASP4

#Add Route 
m3ua route add AS1 700 -1 -1

#Start ASP
m3ua asp start ASP1
m3ua asp start ASP2
m3ua asp start ASP3
m3ua asp start ASP4
